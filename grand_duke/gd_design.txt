    # Extract the text between tildes

#!/bin/bash

# File to process
INPUT_FILE="rqtest.d"

# Backup the original file
if cp "$INPUT_FILE" "$INPUT_FILE.bak"; then
    echo "Backup created: $INPUT_FILE.bak"
else
    echo "Error: Failed to create backup."
    exit 1
fi

# Use awk to process the file and append extracted text
awk '
/^[[:space:]]*SAY ~[^~]+~/ {
    # Check if the line already contains "//"
    if (!/ \/\//) {
        # Extract the text between tildes
        match($0, /~([^~]+)~/, arr)
        if (arr[1] != "") {
            # Append the extracted text with "//" prefix
            $0 = $0 " //" arr[1]
        }
    }
}
{ print }
' "$INPUT_FILE" > temp && mv temp "$INPUT_FILE"

# Check if the operation was successful
if [ $? -eq 0 ]; then
    echo "Processing complete. Changes have been made directly to $INPUT_FILE."
else
    echo "Error: Processing failed. Restoring original file from backup."
    mv "$INPUT_FILE.bak" "$INPUT_FILE"
fi
